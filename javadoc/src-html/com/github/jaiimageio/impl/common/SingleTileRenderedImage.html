045package com.github.jaiimageio.impl.common;
046
047import java.awt.image.ColorModel;
048import java.awt.image.Raster;
049
050/**
051 * A simple class that provides RenderedImage functionality
052 * given a Raster and a ColorModel.
053 */
054public class SingleTileRenderedImage extends SimpleRenderedImage {
055
056    Raster ras;
057
058    /**
059     * Constructs a SingleTileRenderedImage based on a Raster
060     * and a ColorModel.
061     *
062     * @param ras A Raster that will define tile (0, 0) of the image.
063     * @param colorModel A ColorModel that will serve as the image's
064     *           ColorModel.
065     */
066    public SingleTileRenderedImage(Raster ras, ColorModel colorModel) {
067        this.ras = ras;
068
069        this.tileGridXOffset = this.minX = ras.getMinX();
070        this.tileGridYOffset = this.minY = ras.getMinY();
071        this.tileWidth = this.width = ras.getWidth();
072        this.tileHeight = this.height = ras.getHeight();
073        this.sampleModel = ras.getSampleModel();
074        this.colorModel = colorModel;
075    }
076
077    /**
078     * Returns the image's Raster as tile (0, 0).
079     */
080    public Raster getTile(int tileX, int tileY) {
081        if (tileX != 0 || tileY != 0) {
082            throw new IllegalArgumentException("tileX != 0 || tileY != 0");
083        }
084        return ras;
085    }
086}
