047import java.security.AccessController;
048import java.security.PrivilegedAction;
049//import com.sun.medialib.codec.jiio.Util;
050
051public class PackageUtil {
052    /**
053     * Flag indicating whether codecLib is available.
054     */
055    private static boolean isCodecLibAvailable = false;
056
057    /**
058     * Implementation version derived from Manifest.
059     */
060    private static String version = "1.0";
061
062    /**
063     * Implementation vendor derived from Manifest.
064     */
065    private static String vendor = "Sun Microsystems, Inc.";
066
067    /**
068     * Specification Title derived from Manifest.
069     */
070    private static String specTitle = "Java Advanced Imaging Image I/O Tools";
071
072    /**
073     * Set static flags.
074     */
075    static {
076        // Set codecLib flag.
077//        try {
078//            // Check for codecLib availability.
079//            isCodecLibAvailable = Util.isCodecLibAvailable();
080//        } catch(Throwable e) {
081            // A Throwable is equivalent to unavailable. Throwable is used
082            // in case an Error rather than an Exception is thrown.
083            isCodecLibAvailable = false;
084//        }
085
086        // Set version and vendor strings.
087        try {
088            Class thisClass =
089                Class.forName("com.github.jaiimageio.impl.common.PackageUtil");
090            Package thisPackage = thisClass.getPackage();
091            if (thisPackage.getImplementationVersion()!= null && thisPackage.getImplementationVendor() != null) {
092                version = thisPackage.getImplementationVersion();
093                vendor = thisPackage.getImplementationVendor();
094                specTitle = thisPackage.getSpecificationTitle();
095            }
096        } catch(ClassNotFoundException e) {
097        }
098
099      // For some reason, we failed to get this information.
100      if (vendor == null) vendor = "Unknown";
101      if (version == null) version = "Unknown";
102    }
103
104    /**
105     * Returns a <code>boolean</code> indicating whether codecLib is available.
106     */
107    public static final boolean isCodecLibAvailable() {
108        // Retrieve value of system property here to allow this to be
109        // modified dynamically.
110        Boolean result = (Boolean)
111            AccessController.doPrivileged(new PrivilegedAction() {
112                    public Object run() {
113                        String property = null;
114                        try {
115                            property =
116                                System.getProperty("com.github.jaiimageio.disableCodecLib");
117                        } catch(SecurityException se) {
118                            // Do nothing: leave 'property' null.
119                        }
120                        return (property != null &&
121                                property.equalsIgnoreCase("true")) ?
122                            Boolean.TRUE : Boolean.FALSE;
123                    }
124                });
125        boolean isCodecLibDisabled = result.booleanValue();
126
127        return isCodecLibAvailable && !isCodecLibDisabled;
128    }
129
130    /**
131     * Return a version string for the package.
132     */
133    public static final String getVersion() {
134        return version;
135    }
136
137    /**
138     * Return a vendor string for the package.
139     */
140    public static final String getVendor() {
141        return vendor;
142    }
143
144    /**
145     * Return the Specification Title string for the package.
146     */
147    public static final String getSpecificationTitle() {
148        return specTitle;
149    }
150}
